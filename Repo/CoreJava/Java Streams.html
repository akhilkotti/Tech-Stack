1. Employee Model

import java.util.List;

public class Employee {
	private int id;
	private String name;
	private int age;
	private String department;
	private double salary;
	private List<String> skills;
	
	public Employee(int id, String name, int age, String department, 
				   double salary, List<String> skills) {
		this.id = id;
		this.name = name;
		this.age = age;
		this.department = department;
		this.salary = salary;
		this.skills = skills;
	}
	
	public int getId() { return id; }
	public String getName() { return name; }
	public int getAge() { return age; }
	public String getDepartment() { return department; }
	public double getSalary() { return salary; }
	public List<String> getSkills() { return skills; }
	
	public void salaryIncrement(double increment) {
		this.salary = salary + increment;
	}
	
	@Override
	public String toString() {
		return "Employee{" +
			   "name='" + name + '\'' +
			   ", age=" + age +
			   ", department='" + department + '\'' +
			   ", salary=" + salary +
			   ", skills=" + skills +
			   '}' + "\n";
	}
}

33. Flattening a nested list of integers into a single list

↑ Back to Features
JAVA

/*
Working with a flat list is easier when you want to search, filter, or transform data.
Common when processing JSON, CSV, or multidimensional structures (like rows of data).
Each inner list (List<Integer>) becomes a stream of integers.
flatMap() merges all those streams into one continuous stream.
collect() gathers the flattened values into a new list.
*/

List<List<Integer>> nestedList = Arrays.asList(
		Arrays.asList(1, 2),
		Arrays.asList(3, 4, 5),
		Arrays.asList(6)
	);	

// Print the nestedList list
System.out.println("Nested List:\n"+nestedList+"\n");

List<Integer> flatList = 
		// Start a stream from the list of lists (nestedList)
		nestedList.stream()
		
		// Use flatMap to flatten the nested lists into a single stream of integers
		.flatMap(List::stream)                            
		
		// Convert the flattened stream of integers into a List<Integer>
		.collect(Collectors.toList());                   

// Print the flattened list
System.out.println("flatList:\n"+flatList);

65. Calculating weighted average age by department using the teeing collector

↑ Back to Features
JAVA

Map<String, Double> weightedAvgAge = empList.stream()
		.collect(Collectors.groupingBy(
			Employee::getDepartment, // Grouping by department

			// collectingAndThen is used to finalize the result of teeing collector.
			// Collectors.teeing(a, b, merger) is a Java 12 feature that lets you:
				//Run two collectors in parallel (a and b)
				//And combine their results using the merger				        
			Collectors.collectingAndThen(
				Collectors.teeing(
					// Collector 1: Sum of (age × salary) — numerator for weighted average
					Collectors.summingDouble(e -> e.getAge() * e.getSalary()),

					// Collector 2: Sum of salary — denominator for weighted average
					Collectors.summingDouble(Employee::getSalary),

					// Merger: Divide the two to get the weighted average
					(ageSalarySum, totalSalary) -> ageSalarySum / totalSalary
				),

				// Function.identity() just returns the result as-is
				Function.identity()
			)
		));

System.out.println("Weighted average age by department:");
// Print the resulting Map, which contains departments as keys and their weighted average ages as values
System.out.println(weightedAvgAge);

// Create a set to hold intermediate results
Set<String> intermediateResults = new HashSet<>();
listOfLists.stream().forEach(System.out::println);
System.out.print("\n");

// Stream pipeline demonstrating various intermediate operations
List<String> result = listOfLists.stream().flatMap(List::stream) // Flatten the list of lists into a single stream
		.map(String::toUpperCase) // Transform each element to uppercase
		.sorted() // Sort elements
		.filter(s -> s.startsWith("S")) // Filter elements starting with "S"
		.distinct() // Remove duplicate elements
		.peek(s -> intermediateResults.add(s)) // Perform an action (add to set) on each element
		.collect(Collectors.toList()); // Collect the final result into a list

// Print the intermediate results
System.out.println("Intermediate Results:");
System.out.println("Before Sort:");
intermediateResults.forEach(System.out::println);
System.out.print("\n");

System.out.println("After Sort:");
intermediateResults.stream().sorted().forEach(System.out::println);
System.out.print("\n");

// Print the final result
System.out.println("Final Result:");
result.forEach(System.out::println);

// Another example
List<String> topITSkillNames = empList.stream()

	.distinct()                                                           // 9. Remove duplicate skill entries
	.limit(3)                                                             // 10. Limit to top 3 skills
	.collect(Collectors.toList());                                        // 11. Collect as a List
