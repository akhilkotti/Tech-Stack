1. Employee Model

↑ Back to Features
JAVA

import java.util.List;

public class Employee {
	private int id;
	private String name;
	private int age;
	private String department;
	private double salary;
	private List<String> skills;
	
	public Employee(int id, String name, int age, String department, 
				   double salary, List<String> skills) {
		this.id = id;
		this.name = name;
		this.age = age;
		this.department = department;
		this.salary = salary;
		this.skills = skills;
	}
	
	public int getId() { return id; }
	public String getName() { return name; }
	public int getAge() { return age; }
	public String getDepartment() { return department; }
	public double getSalary() { return salary; }
	public List<String> getSkills() { return skills; }
	
	public void salaryIncrement(double increment) {
		this.salary = salary + increment;
	}
	
	@Override
	public String toString() {
		return "Employee{" +
			   "name='" + name + '\'' +
			   ", age=" + age +
			   ", department='" + department + '\'' +
			   ", salary=" + salary +
			   ", skills=" + skills +
			   '}' + "\n";
	}
}

9. Checking if a string or list of strings is empty

↑ Back to Features
JAVA

// Initialize a string variable with an empty string
String str = "";

// Create a stream with the string and check if all elements (only one here) are empty
// allMatch returns true if all elements match the predicate String::isEmpty
boolean isEmpty = Stream.of(str).allMatch(String::isEmpty);

// Print the result based on the boolean outcome
System.out.println(isEmpty ? "String is Empty" : "Not Empty");

// Create a list of strings, including empty, whitespace, and non-empty strings
List strings = List.of("", " ", "abc");

// Check if all strings in the list are empty using Stream API
boolean allEmpty = strings.stream()
		// Returns true only if every string in the list is exactly ""
		.allMatch(String::isEmpty);  

/*
Note: 
- " " (a space) is not considered empty by String::isEmpty, 
  because isEmpty() only returns true for "" (zero-length string).
*/

// Print the result based on the boolean outcome
System.out.println(allEmpty ? "All strings are empty" : "Some strings are not empty");

72. Calculating average age per department

↑ Back to Features
JAVA

// Calculate and print the average age of employees grouped by department
Map avgAgeByDept = empList.stream()

	// Group the employees by department
	.collect(Collectors.groupingBy(
		
		// Key: Department name
		Employee::getDepartment,

		// Downstream collector: average the age of employees in each department
		Collectors.averagingInt(Employee::getAge)
	));

System.out.println("Average age by department:");
// Print the resulting Map, which contains departments as keys and their average ages as values
System.out.println(avgAgeByDept);

73. Finding the specific highest paid employee per department

↑ Back to Features
JAVA

// Create a map of highest paid employee per department
Map highestPaidPerDept = empList.stream()

	// Convert the employee list into a stream for processing
	.collect(Collectors.groupingBy(
		
		// Key: Department name (e.g., "IT", "HR", etc.)
		Employee::getDepartment,

		// Value: Employee with the highest salary in that department
		Collectors.collectingAndThen(

			// First, find the employee with max salary in each department
			Collectors.maxBy(Comparator.comparingDouble(Employee::getSalary)),

			// Since maxBy returns an Optional, we use Optional::get to extract the actual employee object
			Optional::get
		)
	));

System.out.println("Highest paid per department:");
// Print the resulting Map, which contains department names as keys and the highest paid employee in that department as values
System.out.println(highestPaidPerDept);

74. A full demonstration pipeline (Filter -> Sort -> FlatMap -> Map -> Limit)

↑ Back to Features
JAVA

// Create a set to hold intermediate results
Set intermediateResults = new HashSet<>();
listOfLists.stream().forEach(System.out::println);
System.out.print("\n");

// Stream pipeline demonstrating various intermediate operations
List result = listOfLists.stream().flatMap(List::stream) // Flatten the list of lists into a single stream
		.map(String::toUpperCase) // Transform each element to uppercase
		.sorted() // Sort elements
		.filter(s -> s.startsWith("S")) // Filter elements starting with "S"
		.distinct() // Remove duplicate elements
		.peek(s -> intermediateResults.add(s)) // Perform an action (add to set) on each element
		.collect(Collectors.toList()); // Collect the final result into a list

// Print the intermediate results
System.out.println("Intermediate Results:");
System.out.println("Before Sort:");
intermediateResults.forEach(System.out::println);
System.out.print("\n");

System.out.println("After Sort:");
intermediateResults.stream().sorted().forEach(System.out::println);
System.out.print("\n");

// Print the final result
System.out.println("Final Result:");
result.forEach(System.out::println);
System.out.print("\n");

listOfLists.stream().flatMap(List::stream).distinct().collect(Collectors.toList()).forEach(System.out::println);  

System.out.println("Stream pipeline demonstrating multiple operations:\n");

// Another example
List topITSkillNames = empList.stream()
	.filter(e -> e.getDepartment().equals("IT"))                          // 1. Filter only IT employees
	.filter(e -> e.getSalary() > 10000)                                   // 2. Filter those with salary > 85k
	.distinct()                                                           // 3. Remove duplicates (if any)
	.sorted(Comparator.comparing(Employee::getAge))                       // 4. Sort by age ascending
	.peek(e -> System.out.println("After sorting: " + e.getName()))       // 5. Peek to debug stream content
	.flatMap(e -> e.getSkills().stream())                                 // 6. Flatten the list of skills
	.filter(skill -> skill.toLowerCase().contains("java"))                // 7. Filter skills containing "java"
	.map(String::toUpperCase)                                             // 8. Convert to uppercase
	.distinct()                                                           // 9. Remove duplicate skill entries
	.limit(3)                                                             // 10. Limit to top 3 skills
	.collect(Collectors.toList());                                        // 11. Collect as a List

System.out.println("\nFinal Result (Top Java-related IT skills): " + topITSkillNames);
